-- =========================================================
-- 1) TABLES
-- =========================================================

create table if not exists public.branches (
  id uuid primary key default gen_random_uuid(),
  name text not null unique,
  active boolean not null default true,
  created_at timestamptz not null default now()
);

create table if not exists public.staff_profiles (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null unique references auth.users(id) on delete cascade,
  role text not null check (role in ('admin', 'cashier')),
  branch_id uuid references public.branches(id) on delete set null,
  active boolean not null default true,
  created_at timestamptz not null default now()
);

create table if not exists public.gift_campaigns (
  id uuid primary key default gen_random_uuid(),
  name text not null,
  active boolean not null default true,
  start_at timestamptz,
  end_at timestamptz,
  created_at timestamptz not null default now()
);

create table if not exists public.import_batches (
  id uuid primary key default gen_random_uuid(),
  uploaded_by uuid references auth.users(id) on delete set null,
  original_filename text,
  total_rows integer not null default 0,
  valid_rows integer not null default 0,
  invalid_rows integer not null default 0,
  duplicate_rows integer not null default 0,
  suspicious_rows integer not null default 0,
  created_at timestamptz not null default now()
);

create table if not exists public.gift_recipients (
  id uuid primary key default gen_random_uuid(),
  campaign_id uuid references public.gift_campaigns(id) on delete set null,
  import_batch_id uuid references public.import_batches(id) on delete set null,
  phone_raw text,
  phone_normalized text not null,
  gift_type text not null,
  status text not null default 'eligible' check (status in ('eligible', 'claimed', 'redeemed')),
  claimed_at timestamptz,
  redeemed_at timestamptz,
  redeemed_by uuid references auth.users(id) on delete set null,
  branch_id uuid references public.branches(id) on delete set null,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create unique index if not exists gift_recipients_phone_normalized_key
on public.gift_recipients(phone_normalized);

create table if not exists public.gift_qr_sessions (
  id uuid primary key default gen_random_uuid(),
  recipient_id uuid not null references public.gift_recipients(id) on delete cascade,
  token text not null unique,
  expires_at timestamptz not null,
  used_at timestamptz,
  created_at timestamptz not null default now()
);

create table if not exists public.audit_logs (
  id uuid primary key default gen_random_uuid(),
  actor_user_id uuid references auth.users(id) on delete set null,
  action text not null,
  target_table text not null,
  target_id uuid,
  details jsonb not null default '{}'::jsonb,
  created_at timestamptz not null default now()
);

create or replace function public.normalize_phone(input_phone text)
returns text
language plpgsql
immutable
as $$
declare
  cleaned text;
begin
  if input_phone is null then
    return null;
  end if;

  -- keep digits and optional leading +
  cleaned := regexp_replace(trim(input_phone), '[^0-9+]', '', 'g');

  -- Example normalization logic:
  -- 09XXXXXXXX -> +2519XXXXXXXX
  if cleaned ~ '^09[0-9]{8}$' then
    return '+251' || substr(cleaned, 2);
  end if;

  -- 9XXXXXXXX -> +2519XXXXXXXX
  if cleaned ~ '^9[0-9]{8}$' then
    return '+251' || cleaned;
  end if;

  -- 2519XXXXXXXX -> +2519XXXXXXXX
  if cleaned ~ '^2519[0-9]{8}$' then
    return '+' || cleaned;
  end if;

  -- already in +2519XXXXXXXX
  if cleaned ~ '^\+2519[0-9]{8}$' then
    return cleaned;
  end if;

  -- fallback: return cleaned as-is
  return cleaned;
end;
$$;

create or replace function public.is_admin()
returns boolean
language sql
stable
as $$
  select exists (
    select 1
    from public.staff_profiles sp
    where sp.user_id = auth.uid()
      and sp.role = 'admin'
      and sp.active = true
  );
$$;

create or replace function public.is_cashier()
returns boolean
language sql
stable
as $$
  select exists (
    select 1
    from public.staff_profiles sp
    where sp.user_id = auth.uid()
      and sp.role = 'cashier'
      and sp.active = true
  );
$$;

-- 4) RPC: CLAIM GIFT BY PHONE
-- =========================================================

create or replace function public.claim_gift_by_phone(input_phone text)
returns jsonb
language plpgsql
security definer
set search_path = public
as $$
declare
  normalized text;
  recipient_row public.gift_recipients%rowtype;
  qr_token text;
  qr_expiry timestamptz;
  qr_session_id uuid;
begin
  normalized := public.normalize_phone(input_phone);

  if normalized is null or length(normalized) < 10 then
    return jsonb_build_object(
      'success', false,
      'message', 'Invalid phone number.'
    );
  end if;

  select *
  into recipient_row
  from public.gift_recipients
  where phone_normalized = normalized
  limit 1
  for update;

  if not found then
    return jsonb_build_object(
      'success', false,
      'message', 'Phone number not eligible for this promotion.'
    );
  end if;

  if recipient_row.status = 'claimed' or recipient_row.status = 'redeemed' then
    return jsonb_build_object(
      'success', false,
      'message', 'You have already received your gift.'
    );
  end if;

  qr_token := encode(gen_random_bytes(24), 'hex');
  qr_expiry := now() + interval '2 minutes';

  update public.gift_recipients
  set status = 'claimed',
      claimed_at = now()
  where id = recipient_row.id;

  insert into public.gift_qr_sessions (
    recipient_id,
    token,
    expires_at
  )
  values (
    recipient_row.id,
    qr_token,
    qr_expiry
  )
  returning id into qr_session_id;

  insert into public.audit_logs (
    actor_user_id,
    action,
    target_table,
    target_id,
    details
  )
  values (
    null,
    'claim_gift',
    'gift_recipients',
    recipient_row.id,
    jsonb_build_object(
      'phone_normalized', normalized,
      'qr_session_id', qr_session_id,
      'expires_at', qr_expiry
    )
  );

  return jsonb_build_object(
    'success', true,
    'message', 'Gift claimed successfully.',
    'recipient_id', recipient_row.id,
    'gift_type', recipient_row.gift_type,
    'phone_normalized', normalized,
    'qr_token', qr_token,
    'expires_at', qr_expiry
  );
end;
$$;

revoke all on function public.claim_gift_by_phone(text) from public;
grant execute on function public.claim_gift_by_phone(text) to anon, authenticated;

-- =========================================================
-- 5) RPC: REDEEM GIFT BY QR
-- =========================================================

create or replace function public.redeem_gift_by_qr(input_token text)
returns jsonb
language plpgsql
security definer
set search_path = public
as $$
declare
  session_row public.gift_qr_sessions%rowtype;
  recipient_row public.gift_recipients%rowtype;
  staff_row public.staff_profiles%rowtype;
begin
  if auth.uid() is null then
    return jsonb_build_object(
      'success', false,
      'message', 'Authentication required.'
    );
  end if;

  select *
  into staff_row
  from public.staff_profiles
  where user_id = auth.uid()
    and role in ('cashier', 'admin')
    and active = true
  limit 1;

  if not found then
    return jsonb_build_object(
      'success', false,
      'message', 'Only active staff can redeem gifts.'
    );
  end if;

  select *
  into session_row
  from public.gift_qr_sessions
  where token = input_token
  limit 1
  for update;

  if not found then
    return jsonb_build_object(
      'success', false,
      'message', 'Invalid QR code.'
    );
  end if;

  if session_row.used_at is not null then
    return jsonb_build_object(
      'success', false,
      'message', 'This QR code has already been used.'
    );
  end if;

  if session_row.expires_at < now() then
    return jsonb_build_object(
      'success', false,
      'message', 'QR code expired.'
    );
  end if;

  select *
  into recipient_row
  from public.gift_recipients
  where id = session_row.recipient_id
  limit 1
  for update;

  if not found then
    return jsonb_build_object(
      'success', false,
      'message', 'Recipient not found.'
    );
  end if;

  if recipient_row.status = 'redeemed' then
    return jsonb_build_object(
      'success', false,
      'message', 'Gift already redeemed.'
    );
  end if;

  if recipient_row.status <> 'claimed' then
    return jsonb_build_object(
      'success', false,
      'message', 'Gift is not in a redeemable state.'
    );
  end if;

  update public.gift_qr_sessions
  set used_at = now()
  where id = session_row.id;

  update public.gift_recipients
  set status = 'redeemed',
      redeemed_at = now(),
      redeemed_by = auth.uid(),
      branch_id = staff_row.branch_id
  where id = recipient_row.id;

  insert into public.audit_logs (
    actor_user_id,
    action,
    target_table,
    target_id,
    details
  )
  values (
    auth.uid(),
    'redeem_gift',
    'gift_recipients',
    recipient_row.id,
    jsonb_build_object(
      'qr_session_id', session_row.id,
      'branch_id', staff_row.branch_id,
      'cashier_user_id', auth.uid()
    )
  );

  return jsonb_build_object(
    'success', true,
    'message', 'Gift redeemed successfully.',
    'recipient_id', recipient_row.id,
    'gift_type', recipient_row.gift_type,
    'redeemed_at', now()
  );
end;
$$;

revoke all on function public.redeem_gift_by_qr(text) from public;
grant execute on function public.redeem_gift_by_qr(text) to authenticated;

-- =========================================================
-- 6) ADMIN IMPORT HELPER
-- =========================================================

create or replace function public.admin_import_recipient(
  input_batch_id uuid,
  input_phone_raw text,
  input_gift_type text,
  input_campaign_id uuid default null
)
returns jsonb
language plpgsql
security definer
set search_path = public
as $$
declare
  normalized text;
  existing_id uuid;
  new_id uuid;
begin
  if not public.is_admin() then
    return jsonb_build_object(
      'success', false,
      'message', 'Only admins can import recipients.'
    );
  end if;

  normalized := public.normalize_phone(input_phone_raw);

  if normalized is null
     or normalized !~ '^\+2519[0-9]{8}$' then
    return jsonb_build_object(
      'success', false,
      'message', 'Invalid phone format.',
      'phone_raw', input_phone_raw
    );
  end if;

  select id into existing_id
  from public.gift_recipients
  where phone_normalized = normalized
  limit 1;

  if existing_id is not null then
    return jsonb_build_object(
      'success', false,
      'message', 'Duplicate phone number.',
      'phone_normalized', normalized,
      'existing_id', existing_id
    );
  end if;

  insert into public.gift_recipients (
    campaign_id,
    import_batch_id,
    phone_raw,
    phone_normalized,
    gift_type,
    status
  )
  values (
    input_campaign_id,
    input_batch_id,
    input_phone_raw,
    normalized,
    input_gift_type,
    'eligible'
  )
  returning id into new_id;

  return jsonb_build_object(
    'success', true,
    'message', 'Recipient imported.',
    'recipient_id', new_id,
    'phone_normalized', normalized
  );
end;
$$;

revoke all on function public.admin_import_recipient(uuid, text, text, uuid) from public;
grant execute on function public.admin_import_recipient(uuid, text, text, uuid) to authenticated;
